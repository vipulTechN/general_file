Gallery section js

var swiper = new Swiper(".mySwiper", {
  spaceBetween: 30,
  autoplay: {
    delay: 1500,
    disableOnInteraction: false,
  },
  slidesPerView: 2,
  grabCursor: true,
  keyboard: {
    enabled: true,
  },
  breakpoints: {
    250: {
      slidesPerView: 1,
    },
    700: {
      slidesPerView: 2,
    }, 
    1650: {
      slidesPerView: 3,
    },

  },
  pagination: {
    el: ".swiper-pagination",
    clickable: true,
  },
  navigation: {
    nextEl: ".swiper-button-next",
    prevEl: ".swiper-button-prev",
  },
});
// Get all elements with class "myImg"
var images = document.getElementsByClassName('myImgswiper');

// Get the modal
var modal = document.getElementById('myModalswiper');
var modalImg = document.getElementById("imgswiper");

// Loop through all images and attach click event handler
for (var i = 0; i < images.length; i++) {
    images[i].onclick = function() {
        modal.style.display = "block";
        modalImg.src = this.src;
        captionText.innerHTML = this.alt;
    };
}

// Get the <span> element that closes the modal
var span = document.getElementsByClassName("closemodal")[0];

// When the user clicks on <span> (x), close the modal
span.onclick = function() { 
    modal.style.display = "none";
};
