<script>
        document.addEventListener('DOMContentLoaded', () => {
            const filterButtons = document.querySelectorAll('.filter-btn');
            const items = document.querySelectorAll('.item');

            filterButtons.forEach(button => {
                button.addEventListener('click', () => {
                    document.querySelector('.filter-btn.activeblog').classList.remove('activeblog');
                    button.classList.add('activeblog');
                    filterItems(button.getAttribute('data-filter'));
                });
            });

            const filterItems = (filter) => {
                items.forEach(item => {
                    item.classList.toggle('hidden', filter !== 'all' && !item.classList.contains(filter));
                });
            };
        });

    </script>
